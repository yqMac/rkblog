# 常用命令

以下枚举值均可通过--help查询详细字典

# -m 1420: 加密方式 sha256($salt.$pass)
# -a 6: 解密方式 6：Hybrid Wordlist + Mask ： 字典后加掩码 
# region_1950_2020.dict 使用的字典文件
# --custom-charset1 ?dX 自定义字符集(整型+X)
# -o salt_test.out 结果输出到 salt_test.out 文件
# --status --status-timer 10  10s自动打印状态信息

# 身份证
nohup hashcat -m 1420 myhash01.txt -a 6 region_1950_2020.dict --custom-charset1 ?dX ?d?d?d?1 --force -o salt_test.out --status --status-timer 10 >> hashcat.log 2>&1 &
# 手机号
nohup hashcat -m 1420 20211027_phone.txt -a 6 region.txt  ?1?d?d?d?d?d?d?d?d --force -o mobile_20211027.out --status --status-timer 10 >> hashcat.log 2>&1 &
